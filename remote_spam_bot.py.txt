import discord
from discord.ext import commands
from discord import app_commands
import os
from dotenv import load_dotenv
import asyncio
import aiohttp

load_dotenv()
TOKEN = os.getenv('DISCORD_TOKEN')

if not TOKEN:
    print("❌ DISCORD_TOKEN not found in .env file!")
    exit(1)

intents = discord.Intents.default()
intents.message_content = True
intents.members = True
intents.guilds = True

bot = commands.Bot(command_prefix='!', intents=intents)

@bot.event
async def on_ready():
    print(f'✅ {bot.user} logged in')
    print(f'✅ Bot is in {len(bot.guilds)} servers')
    try:
        synced = await bot.tree.sync()
        print(f"✅ {len(synced)} commands synced")
    except Exception as e:
        print(f"❌ Sync error: {e}")

# ============== WEBHOOK SPAM ==============

@bot.tree.command(name="webhook_spam", description="Fast spam using webhook")
@app_commands.describe(
    webhook_url="Discord webhook URL",
    message="Message to spam",
    count="Number of times (Max 50)"
)
async def webhook_spam(
    interaction: discord.Interaction,
    webhook_url: str,
    message: str,
    count: int = 10
):
    """⚡ Spam via webhook"""
    
    if count > 50:
        await interaction.response.send_message("❌ Max 50!", ephemeral=True)
        return
    
    if not webhook_url.startswith("https://discord.com/api/webhooks/"):
        await interaction.response.send_message("❌ Invalid webhook URL!", ephemeral=True)
        return
    
    await interaction.response.defer(ephemeral=True)
    
    try:
        async with aiohttp.ClientSession() as session:
            success = 0
            
            for i in range(count):
                try:
                    async with session.post(
                        webhook_url,
                        json={"content": message},
                        timeout=aiohttp.ClientTimeout(total=5)
                    ) as resp:
                        if resp.status in [200, 204]:
                            success += 1
                    await asyncio.sleep(0.3)
                except Exception as e:
                    print(f"Error: {e}")
            
            embed = discord.Embed(
                title="✅ Complete!",
                description=f"Sent: **{success}/{count}**",
                color=discord.Color.green()
            )
            await interaction.followup.send(embed=embed, ephemeral=True)
    
    except Exception as e:
        await interaction.followup.send(f"❌ Error: {e}", ephemeral=True)

# ============== HELP COMMAND ==============

@bot.command(name='help')
async def help_cmd(ctx):
    """Bot help"""
    embed = discord.Embed(
        title="🤖 Spam Bot Help",
        description="Use `/webhook_spam` to spam messages via webhook",
        color=discord.Color.blue()
    )
    await ctx.send(embed=embed)

# Run bot
if __name__ == "__main__":
    bot.run(TOKEN)
