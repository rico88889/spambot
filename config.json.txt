# ============== WEBHOOK SPAM (FAST) ==============

@bot.tree.command(name="webhook_spam", description="Fast spam using webhook")
@app_commands.describe(
    webhook_url="Discord webhook URL",
    message="Message to spam",
    count="Number of times (Max 100)"
)
async def webhook_spam(
    interaction: discord.Interaction,
    webhook_url: str,
    message: str,
    count: int = 10
):
    """⚡ FASTEST SPAM METHOD - Using Webhooks"""
    
    # Validation
    if count > Config.Spam.MAX_SPAM_COUNT:
        await interaction.response.send_message(
            get_error_message('max_spam', max=Config.Spam.MAX_SPAM_COUNT),
            ephemeral=True
        )
        return
    
    if count < 1:
        await interaction.response.send_message("❌ Minimum 1 time!", ephemeral=True)
        return
    
    if len(message) > Config.Spam.MAX_MESSAGE_LENGTH:
        await interaction.response.send_message(
            get_error_message('max_length', max=Config.Spam.MAX_MESSAGE_LENGTH),
            ephemeral=True
        )
        return
    
    # Validate webhook URL
    if not webhook_url.startswith("https://discord.com/api/webhooks/"):
        await interaction.response.send_message("❌ Invalid webhook URL!", ephemeral=True)
        return
    
    await interaction.response.defer(ephemeral=True)
    
    try:
        async with aiohttp.ClientSession() as session:
            data = {
                "content": Config.Invite.INVITE_MESSAGE + "\n\n" + message,
                "username": "Spam Bot"
            }
            
            success = 0
            failed = 0
            
            for i in range(count):
                try:
                    async with session.post(webhook_url, json=data, timeout=aiohttp.ClientTimeout(total=5)) as resp:
                        if resp.status == 204 or resp.status == 200:
                            success += 1
                        else:
                            failed += 1
                    await asyncio.sleep(Config.Spam.WEBHOOK_DELAY)
                except asyncio.TimeoutError:
                    failed += 1
                except Exception as e:
                    failed += 1
                    print(f"❌ Error: {e}")
            
            embed = discord.Embed(
                title="⚡ Webhook Spam Complete!",
                description=f"✅ Sent: **{success}**\n❌ Failed: **{failed}**",
                color=Config.Color.SUCCESS
            )
            await interaction.followup.send(embed=embed, ephemeral=True)
    
    except Exception as e:
        await interaction.followup.send(f"❌ Error: {str(e)[:100]}", ephemeral=True)

# ============== MULTI-WEBHOOK SPAM (ULTRA FAST) ==============

@bot.tree.command(name="multi_webhook_spam", description="Spam multiple webhooks at once")
@app_commands.describe(
    webhooks="Webhook URLs separated by comma",
    message="Message to spam",
    count="Number of times per webhook"
)
async def multi_webhook_spam(
    interaction: discord.Interaction,
    webhooks: str,
    message: str,
    count: int = 5
):
    """⚡⚡ ULTRA FAST - Spam multiple webhooks simultaneously"""
    
    if count > 30:
        await interaction.response.send_message("❌ Max 30 per webhook!", ephemeral=True)
        return
    
    webhook_list = [w.strip() for w in webhooks.split(",")]
    
    await interaction.response.defer(ephemeral=True)
    
    try:
        async with aiohttp.ClientSession() as session:
            tasks = []
            
            for webhook_url in webhook_list:
                for i in range(count):
                    data = {
                        "content": Config.Invite.INVITE_MESSAGE + "\n\n" + message,
                        "username": "Spam Bot"
                    }
                    task = session.post(webhook_url, json=data, timeout=aiohttp.ClientTimeout(total=5))
                    tasks.append(task)
            
            # Execute all at once
            results = await asyncio.gather(*tasks, return_exceptions=True)
            
            success = sum(1 for r in results if not isinstance(r, Exception))
            failed = len(results) - success
            
            embed = discord.Embed(
                title="⚡⚡ Multi-Webhook Spam Complete!",
                description=f"✅ Sent: **{success}**\n❌ Failed: **{failed}**\n📊 Webhooks: **{len(webhook_list)}**",
                color=Config.Color.SUCCESS
            )
            await interaction.followup.send(embed=embed, ephemeral=True)
    
    except Exception as e:
        await interaction.followup.send(f"❌ Error: {e}", ephemeral=True)

# ============== REMOTE CHANNEL SPAM (FAST) ==============

@bot.tree.command(name="remotespam", description="Spam channels (bot must be in server)")
@app_commands.describe(
    server_id="Target server ID",
    message="Message to spam",
    count="Number of times (Max 50)",
    channel_id="Specific channel ID (optional)"
)
async def remote_spam(
    interaction: discord.Interaction,
    server_id: str,
    message: str,
    count: int = 10,
    channel_id: str = None
):
    """Spam channels in a server"""
    
    if count > Config.Spam.MAX_SPAM_COUNT:
        await interaction.response.send_message(
            get_error_message('max_spam', max=Config.Spam.MAX_SPAM_COUNT),
            ephemeral=True
        )
        return
    
    await interaction.response.defer(ephemeral=True)
    
    try:
        try:
            target_guild = bot.get_guild(int(server_id))
        except ValueError:
            await interaction.followup.send("❌ Server ID must be a number!", ephemeral=True)
            return
        
        if not target_guild:
            await interaction.followup.send(
                f"❌ Server not found!\n**Your server ID:** `{interaction.guild.id}`",
                ephemeral=True
            )
            return
        
        # Select channels
        channels_to_spam = []
        
        if channel_id:
            try:
                target_channel = target_guild.get_channel(int(channel_id))
                if not target_channel:
                    await interaction.followup.send("❌ Channel not found!", ephemeral=True)
                    return
                channels_to_spam = [target_channel]
            except ValueError:
                await interaction.followup.send("❌ Invalid channel ID!", ephemeral=True)
                return
        else:
            channels_to_spam = [
                ch for ch in target_guild.text_channels
                if isinstance(ch, discord.TextChannel) and 
                ch.permissions_for(target_guild.me).send_messages
            ]
        
        if not channels_to_spam:
            await interaction.followup.send(
                "❌ Bot has no permission to send messages!",
                ephemeral=True
            )
            return
        
        spam_message = Config.Invite.INVITE_MESSAGE + "\n\n" + message
        
        success = 0
        failed = 0
        
        for channel in channels_to_spam:
            for i in range(count):
                try:
                    await channel.send(spam_message)
                    success += 1
                    await asyncio.sleep(Config.Spam.MESSAGE_DELAY)
                except discord.Forbidden:
                    failed += 1
                    break
                except Exception as e:
                    failed += 1
                    break
        
        embed = discord.Embed(
            title="✅ Remote Spam Complete!",
            description=f"Server: **{target_guild.name}**\n" +
                       f"Channels: **{len(channels_to_spam)}**\n" +
                       f"Sent: **{success}**\n" +
                       f"Failed: **{failed}**",
            color=Config.Color.SUCCESS
        )
        await interaction.followup.send(embed=embed, ephemeral=True)
    
    except Exception as e:
        await interaction.followup.send(f"❌ Error: {e}", ephemeral=True)
